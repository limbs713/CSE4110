CREATE TABLE Shipping (package_id INTEGER NOT NULL, shipping_id INTEGER NOT NULL, transport_id INTEGER NOT NULL, location VARCHAR(30) NOT NULL, destination VARCHAR(30) NOT NULL, leave_location_time TIMESTAMP NULL check((YEAR(leave_location_time) > 2000)), arrive_destination_time TIMESTAMP NULL check((YEAR(arrive_destination_time) > 2000)), primary key (shipping_id), foreign key (package_id) references Package (package_id) on delete cascade, foreign key (transport_id) references transport (transport_id) on delete cascade);
CREATE TABLE Bill (bill_id INTEGER NOT NULL, sender_id INTEGER NOT NULL, receiver_id INTEGER NOT NULL, package_id INTEGER NOT NULL, order_type VARCHAR(30) NOT NULL check (order_type in ('purchase', 'refund')), sign_date TIMESTAMP NOT NULL check((YEAR(sign_date) > 2000)), cost INTEGER NOT NULL, primary key (bill_id), foreign key (sender_id) references Customer (customer_id) on delete cascade, foreign key (receiver_id) references Customer (customer_id) on delete cascade, foreign key (package_id) references Package (package_id) on delete cascade);
CREATE TABLE amount_owed (user_id INTEGER NOT NULL, amount BIGINT NOT NULL, primary key (user_id), foreign key (user_id) references Customer (customer_id) on delete cascade);
